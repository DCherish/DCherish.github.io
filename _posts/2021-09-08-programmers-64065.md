```c++
#include <string>
#include <vector>
#include <sstream>
#include <algorithm>
#include <map>

using namespace std;

struct info
{
	int cnt;
	int idx;
};

map<int, int> MAP;

bool cmp(info a, info b)
{
	return a.cnt > b.cnt;
}

vector<int> solution(string s)
{
	vector<int> answer;

	stringstream ss(s);
	string token;

	while (getline(ss, token, '}'))
	{
		if (token == "") break;

		string str = "";

		for (int i = 2; i < token.length(); i++)
		{
			if (token[i] >= '0' && token[i] <= '9')
			{
				str += token[i];
			}
			else if (token[i] == ',')
			{
				MAP[stoi(str)]++;
                
                str = "";
			}
		}

		MAP[stoi(str)]++;
	}

	vector<info> vec;

	for (int i = 1; i <= 100000; i++)
	{
		if (MAP.count(i) != 0)
		{
			vec.push_back({ MAP[i], i });
		}
	}

	sort(vec.begin(), vec.end(), cmp);

	for (int i = 0; i < vec.size(); i++)
	{
		answer.push_back(vec[i].idx);
	}

	return answer;
}
```
