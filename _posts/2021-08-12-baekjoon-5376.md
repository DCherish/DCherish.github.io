#include <iostream>
#include <string>
#include <vector>

using namespace std;
typedef long long ll;

int T;

vector<string> ans;

ll gcd(ll x, ll y)
{
    if (x < y)
    {
        ll temp = x;
        ll x = y;
        ll y = temp;
    }
    
	while (true)
	{
		if (x % y == 0) return y;
		else
		{
			ll temp = x % y;
			x = y;
			y = temp;
		}
	}
}
