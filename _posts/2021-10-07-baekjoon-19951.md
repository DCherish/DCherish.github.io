int N, M, a, b, k, idx;

	for (int i = 1; i <= N; i++)
	{
		cin >> arr[i];
	}

	for (int i = 0; i < M; i++)
	{
		cin >> a >> b >> k;

		cmd[a] += k;
		cmd[b + 1] -= k;
	}

	{
		cmd[i] += cmd[i - 1];

		cout << arr[i] + cmd[i] << " ";
	}

	cout << "\n";

	return 0;
}
